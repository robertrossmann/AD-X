 21 /**
 22  * This class is responsible for managing the directory schema
 23  *
 24  * @see         <a href="http://msdn.microsoft.com/en-us/library/windows/desktop/ms674984%28v=vs.85%29.aspx">MSDN - Active Directory Schema</a>
 25  */
 26 class Schema
 27 {
 28     /**
 29      * Contains schema objects loaded from files for fast runtime access
 30      *
 31      * This ensures that if multiple objects request the same attribute schema,
 32      * the Schema class will have to load that file only once and then it will
 33      * serve that data from memory.
 34      *
 35      * @var array
 36      */
 37     protected static $runtime_cache = array();
 38 
 39     /**
 40      * Folder where the schema is going to be stored.
 41      *
 42      * This folder will <b>always</b> be relative to the Schema.php file.
 43      *
 44      * @var         string
 45      */
 46     protected static $schema_dir = 'Schema';
 47 
 48     /**
 49      * These attributes will be loaded about the schema object
 50      *
 51      * @var         array
 52      */
 53     protected static $attribute_properties = [
 54         'ldapdisplayname',
 55         'attributesyntax',
 56         'omsyntax',
 57         'issinglevalued',
 58         'rangelower',
 59         'rangeupper',
 60         'systemflags',
 61     ];
 62 
 63     protected static $class_properties = [
 64         'ldapdisplayname',
 65         'rdnattid',
 66         'subclassof',
 67         'allowedattributes',
 68         'systemonly',
 69     ];
 70 
 71     final private function __construct() {}
 72

 74     /**
 75      * Is the Directory schema currently cached?
 76      *
 77      * @return      boolean     A boolean value identifying the presence or absence of the schema cache
 78      */
 79     public static function isCached()
 80     {
 81         return file_exists( ADX_ROOT_PATH . static::$schema_dir . ADX_DS . '.lockfile' ) ? true : false;
 82     }
 83 
 84     /**
 85      * Build the local schema from server, using provided {@link Link}
 86      *
 87      * This method creates the locally cached schema from all schema objects located in
 88      * CN=Schema,CN=Configuration ( or similar, depending on domain configuration ).
 89      * The location of the Schema is taken from the RootDSE's "schemanamingcontext" entry.
 90      *
 91      * @param       Link        The Link object to be used to connect to directory server
 92      *
 93      * @return      void
 94      */
 95     public static function build( Link $adxLink )
 96     {
 97         // Define where to store the schema definition
 98         $schemaDir = ADX_ROOT_PATH . static::$schema_dir;
 99 
100         // Prepare the schema folder either by cleaning it's contents or by creating it
101         file_exists( $schemaDir ) ? static::flush() : mkdir( $schemaDir, 0755 );
102 
103         $schema_base = $adxLink->rootDSE->schemaNamingContext(0); // schemanamingcontext is loaded by default
104 
105         // Create the tasks...
106         // I have to create them separately because I have two different
107         // sets of attributes that I need to have loaded
108         $tasks[0] = new Task( Enums\Operation::OpList, $adxLink );
109         $tasks[0]   ->use_pages( 500 )
110                     ->base( $schema_base )
111                     ->filter( q::a( ['objectclass' => 'attributeschema'] ) )        // Attribute definitions
112                     ->attributes( static::$attribute_properties );
113 
114         $tasks[1] = new Task( Enums\Operation::OpList, $adxLink );
115         $tasks[1]   ->use_pages( 500 )
116                     ->base( $schema_base )
117                     ->filter( q::a( ['objectclass' => 'classschema'] ) )            // Class definitions
118                     ->attributes( static::$class_properties );
119 
120         // And retrieve the schema objects!
121         foreach ( $tasks as $task )
122         {
123             // Do not use the Task::run_paged() method as it will very likely hit the memory execution limit.
124             // Instead, mimick that method's functionality and handle the data for each page separately
125             do
126             {
127                 $objects = $task->run();
128 
129                 if ( $objects )
130                 {
131                     // Loop through the schema objects and save them to a local file, named
132                     // after the attribute they represent
133                     foreach ( $objects as $object )
134                     {
135                         $filename = $object->ldapDisplayName(0) . ".json";
136                         $data = $object->json();
137 
138                         file_put_contents( $schemaDir. ADX_DS . strtolower( $filename ), $data );
139                     }
140                 }
141                 else throw new Exception( 'Maximum number of referrals reached' );
142             }
143             while ( ! $task->complete );
144         }
145 
146         // Generate a lockfile to identify the fact that the Schema is present
147         file_put_contents( $schemaDir . ADX_DS . '.lockfile', time() );
148     }
149 
150     /**
151      * Clear the whole Schema cache folder
152      *
153      * This method clears all data from the Schema cache folder.
154      *
155      * @return      void
156      */
157     public static function flush()
158     {
159         array_map( 'unlink', glob( ADX_ROOT_PATH . static::$schema_dir . ADX_DS . '*.json' ) );
160     }

162     /**
163      * Get the cached data about a schema object
164      *
165      * Use this function to get the data that is cached in the Schema cache
166      * for a specified attribute or object.
167      *
168      * @param       string      ldap name of the attribute / object you want the Schema data for
169      *
170      * @return      array|null  The Schema data for the specified attribute / object or null if that object is not present in the schema cache
171      */
172     public static function get( $schema_object )
173     {
174         // Convert to lowercase...
175         $schema_object = strtolower( $schema_object );
176 
177         // Check if this schema object is already loaded in the runtime cache and return it if so
178         if ( array_key_exists( $schema_object, static::$runtime_cache ) ) return static::$runtime_cache[$schema_object];
179 
180         // No, it is not - load it from the file and store it in runtime cache for future re-use
181 
182         $schema_file = ADX_ROOT_PATH . static::$schema_dir . ADX_DS . "$schema_object.json";
183 
184         // Check if this file has been cached and load it if so
185         if ( file_exists( $schema_file ) )
186         {
187             // Got the schema file, so let's load it and read the properties
188             $json   = file_get_contents( $schema_file );
189             $data   = json_decode( $json, true, 512, JSON_BIGINT_AS_STRING );
190 
191             // Store the data in runtime cache
192             static::$runtime_cache[$schema_object] = $data;
193 
194             return $data;
195         }
196         else return null;   // This schema object is not present in local schema cache - nothing to return...
197     }
